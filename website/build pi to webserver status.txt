instructions to setup
turn on ssh
sudo systemctl enable ssh
sudo systemctl start ssh
sudo apt-get install software-properties-common
sudo apt-get install openssh server


adduser sam 
usermod -aG sudo sam

ssh-keygen -t rsa -b 4096 -f sam_key
cat sam_key.pub >>authorized_keys
scp sam@192.168.118.65:/home/sam/.ssh/sam_key ./.ssh/sam_key

sudo nano /etc/ssh/sshd_config
uncomment passwordauthentication yes 
change yes to no
save and exit
sudo service ssh restart

create config and edit
 
sudo nano /etc/hostname
change to samuelwarrenconsulting.com
sudo nano /etc/hosts
change 127.0.0.1 to samuelwarrenconsulting.com


sudo nano /boot/config.txt

scroll to bottom and add 
#overclock lines
arm_freq=1800 (1.8ghz)
over_voltage=2

sudo shutdown -r now

shutdown bluetooth
shutdown wifi

sudo nano /boot/config.txt

#bluetooth and wifi disable
dtoverlay=pi3-disable-bt
dtoverlay=pi3-disable-wifi

--disable xwindow on boot
and bluetooth service--

sudo raspi-config
1 system options s7 splash screen = no
s5 set to console

sudo systemctl disable bluetooth.service
sudo systemctl disable bluealsa.service
sudo systemctl disable hciuart.service

curl -fsSL https://get.docker.com -o get-docker.sh

sudo sh get-docker.sh
sudo usermod -aG docker ${USER}
